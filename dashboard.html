// ============================================ // 🆕 SALES DATA FUNCTIONS - Add this to the script section // ============================================ async function refreshSalesData() { console.log('🔄 Refreshing sales data...'); try { // Fetch both state and latest report const [stateRes, reportRes] = await Promise.all([ fetch('/api/sales/state'), fetch('/api/sales/latest') ]); const stateData = await stateRes.json(); const reportData = await reportRes.json(); // Update stats cards with numbers document.getElementById('totalLeads').textContent = stateData.performance?.totalLeads || 0; document.getElementById('totalOutreach').textContent = stateData.performance?.totalOutreach || 0; document.getElementById('totalConversions').textContent = stateData.performance?.totalConversions || 0; document.getElementById('totalRevenue').textContent = '$' + (stateData.performance?.totalRevenue || 0).toLocaleString(); document.getElementById('interestedLeads').textContent = reportData.summary?.interested || 0; // Update the response box with full summary const responseBox = document.getElementById('responseBox'); if (reportData.summary && reportData.summary.leadsDiscovered > 0) { responseBox.innerHTML = ` 📊 Sales Report (${reportData.date || 'Today'})
🟢 Leads: ${reportData.summary.leadsDiscovered}
📨 Outreach: ${reportData.summary.outreachSent}
🎯 Interested: ${reportData.summary.interested}
✅ Conversions: ${reportData.summary.conversions}
💰 Revenue: $${reportData.summary.revenue}
📈 Rate: ${reportData.summary.conversionRate}
${reportData.nextSteps ? `
Next Steps:
    ${reportData.nextSteps.map(step => `
  • ${step}
    • `).join('')}
` : ''}
`; } else { responseBox.innerHTML = ` 💡 No sales data yet. Run your Autonomous Sales Agent to start generating leads!

💡 Run: node autonomous-sales-agent.js in the slideshow-kit folder `; } // Show sales summary section const salesSummary = document.getElementById('salesSummary'); salesSummary.style.display = 'block'; salesSummary.innerHTML = ` 📈 Daily Performance:
📊 Conversion Rate: ${reportData.summary?.conversionRate || 'N/A'}
💰 Revenue per Lead: $${reportData.summary?.leadsDiscovered > 0 ? Math.round(reportData.summary.revenue / reportData.summary.leadsDiscovered) : 0}
🎯 Interest Rate: ${reportData.summary?.outreachSent > 0 ? Math.round(reportData.summary.interested / reportData.summary.outreachSent * 100) : 0}%
`; } catch (error) { console.error('❌ Error fetching sales data:', error); document.getElementById('responseBox').innerHTML = '❌ Error loading sales data: ' + error.message; } } function viewSalesReport() { window.open('/api/sales/latest', '_blank'); } // Automatically refresh sales data when page loads // Call this after the existing DOMContentLoaded event setTimeout(refreshSalesData, 1500);
