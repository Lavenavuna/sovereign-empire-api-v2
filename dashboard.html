async function refreshStatus() { const timestamp = document.getElementById('statusTimestamp'); if (timestamp) timestamp.textContent = new Date().toLocaleTimeString(); const checks = [ { id: 'statusRailway', url: '/health', label: 'Railway' }, { id: 'statusApi', url: '/health', label: 'API' }, { id: 'statusSales', url: '/api/leads/stats', label: 'Sales' }, { id: 'statusAgents', url: '/api/agents', label: 'Agents' }, ]; let allOnline = true; for (const check of checks) { const el = document.getElementById(check.id); if (!el) continue; try { const res = await fetch(check.url); if (res.ok) { if (check.id === 'statusSales') { const data = await res.json(); const count = data.stats?.total || 0; el.textContent = '✅ Online (' + count + ' leads)'; el.className = 'badge status-online'; } else if (check.id === 'statusAgents') { const data = await res.json(); const count = data.count || 0; el.textContent = '✅ Online (' + count + ' agents)'; el.className = 'badge status-online'; } else { el.textContent = '✅ Online'; el.className = 'badge status-online'; } } else { el.textContent = '⚠️ Degraded'; el.className = 'badge status-degraded'; allOnline = false; } } catch (e) { el.textContent = '❌ Offline'; el.className = 'badge status-offline'; allOnline = false; } } // Bloom AI - check via test const bloomEl = document.getElementById('statusBloom'); if (bloomEl) { try { const res = await fetch('/health'); if (res.ok) { bloomEl.textContent = '✅ Active'; bloomEl.className = 'badge status-online'; } else { bloomEl.textContent = '⚠️ Degraded'; bloomEl.className = 'badge status-degraded'; allOnline = false; } } catch (e) { bloomEl.textContent = '❌ Offline'; bloomEl.className = 'badge status-offline'; allOnline = false; } } // Scheduler - check via health const schedEl = document.getElementById('statusScheduler'); if (schedEl) { try { const res = await fetch('/health'); if (res.ok) { schedEl.textContent = '✅ Running'; schedEl.className = 'badge status-online'; } else { schedEl.textContent = '⚠️ Degraded'; schedEl.className = 'badge status-degraded'; allOnline = false; } } catch (e) { schedEl.textContent = '❌ Offline'; schedEl.className = 'badge status-offline'; allOnline = false; } } // Overall status const overall = document.getElementById('overallStatus'); if (overall) { if (allOnline) { overall.textContent = '🟢 All Systems Operational'; overall.className = 'status-badge status-online'; } else { overall.textContent = '🟡 Some Systems Degraded'; overall.className = 'status-badge status-degraded'; } } }
